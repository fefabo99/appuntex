\begin{equation*}
    F_g = G\frac{m_1m_2}{r^2} \; \Bigg[\frac{m^3}{\text{Kg}\cdot s^2} \cdot
    \frac{\text{Kg}^2}{m^2} = \frac{m \cdot \text{Kg}}{s^2} \Bigg]
\end{equation*}
Dove $G$ è la costante di gravitazione universale:
\begin{equation*}
    G = 6,673^{-11} \;\Bigg[\frac{m^3}{\text{Kg}\cdot s^2}\Bigg]
\end{equation*}
